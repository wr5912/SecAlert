       ```python
       @router.post("/preview-parse")
       async def preview_parse(request: PreviewParseRequest):
           """预览解析结果 - 使用真实 ThreeTierParser"""
           # 获取模板
           template = _templates.get(request.template_id)
           if not template:
               raise HTTPException(status_code=404, detail="Template not found")

           parser = ThreeTierParser()
           results = []

           for log in request.logs:
               try:
                   parsed = parser.parse(log, template.get("source_type", "unknown"))
                   results.append({
                       "raw": log,
                       "parsed": parsed,
                       "status": "success"
                   })
               except Exception as e:
                   results.append({
                       "raw": log,
                       "parsed": None,
                       "status": "failure",
                       "error": str(e)
                   })

           return {
               "results": results,
               "template_id": request.template_id
           }
       ```

       ```python
       @router.post("/templates/batch")
       async def batch_create_templates(request: BatchCreateRequest):
           results = []
           success_count = 0
           failure_count = 0

           for device in request.devices:
               try:
                   # 确定使用的日志格式和正则
                   if request.apply_template_id:
                       template = _templates.get(request.apply_template_id)
                       if template:
                           log_format = template.get("log_format", "Auto")
                           custom_regex = template.get("custom_regex")
                       else:
                           log_format = device.log_format
                           custom_regex = None
                   else:
                       log_format = device.log_format
                       custom_regex = None

                   # 创建模板
                   template_data = {
                       "name": device.name,
                       "device_type": device.device_type,
                       "connection": {
                           "host": device.host,
                           "port": device.port,
                           "protocol": device.protocol
                       },
                       "log_format": log_format,
                       "custom_regex": custom_regex
                   }
                   template = create_template(template_data)
                   results.append({
                       "name": device.name,
                       "template_id": template.get("id"),
                       "status": "success",
                       "error": None
                   })
                   success_count += 1
               except Exception as e:
                   results.append({
                       "name": device.name,
                       "template_id": None,
                       "status": "failure",
                       "error": str(e)
                   })
                   failure_count += 1

           return {
               "success_count": success_count,
               "failure_count": failure_count,
               "results": results
           }
       ```
