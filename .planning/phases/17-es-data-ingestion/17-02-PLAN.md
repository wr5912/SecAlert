**新的 elasticsearch input query:**
```ruby
query => '{"query": {
  "bool": {
    "must": [
      { "range": { "@timestamp": { "gte": "now-1h", "lte": "now" } } }
    ],
    "should": [
      { "terms": { "alert.severity": ["high", "critical"] } },
      { "terms": { "severity": ["high", "critical"] } },
      { "range": { "risk_score": { "gte": 70 } } }
    ],
    "minimum_should_match": 1
  }
}}'
```

说明：
- `range @timestamp` 限制只拉取最近 1 小时的告警（避免重复拉取）
- `should` 条件匹配高危告警：alert.severity、severity、risk_score
- `minimum_should_match: 1` 确保至少匹配一个条件
- 这是一个占位符查询，实际部署时可根据 ES 中的实际字段名调整

  <name>Task 2: 添加告警字段映射规则</name>
  <files>logstash/pipeline/es-input.conf</files>
  <read_first>logstash/pipeline/es-input.conf</read_first>
  <action>
在 filter 部分添加字段映射规则，将 ES 中的原始字段映射为统一格式。

**在现有 date filter 之后添加以下 filter 块:**

```ruby
  # ========== 告警字段映射 ==========
  # 使用 dissect 替代 Grok（性能 5-10 倍，D-08）

  # 1. 告警级别标准化
  if [alert][severity] or [severity] {
    mutate {
      add_field => { "alert_level" => "%{[alert][severity]:%{[severity]:unknown}}" }
    }
    ruby {
      code => "
        level = event.get('alert_level')
        level_map = { 'low' => 1, 'medium' => 2, 'high' => 3, 'critical' => 4, 'unknown' => 0 }
        event.set('severity_id', level_map.fetch(level, 0))
      "
    }
  }

  # 2. 告警标题和描述
  if [alert][title] or [title] {
    mutate {
      add_field => { "alert_title" => "%{[alert][title]:%{[title]}}" }
    }
  }

  if [alert][description] or [description] {
    mutate {
      add_field => { "alert_desc" => "%{[alert][description]:%{[description]}}" }
    }
  }

  # 3. 源目 IP 提取（网络告警常用）
  if [src][ip] or [source][ip] or [srcip] {
    mutate {
      add_field => { "src_ip" => "%{[src][ip]:%{[source][ip]:%{[srcip]}}}" }
    }
  }

  if [dst][ip] or [dest][ip] or [dstip] {
    mutate {
      add_field => { "dst_ip" => "%{[dst][ip]:%{[dest][ip]:%{[dstip]}}}" }
    }
  }

  # 4. 端口提取
  if [src][port] or [source][port] or [srcport] {
    mutate {
      add_field => { "src_port" => "%{[src][port]:%{[source][port]:%{[srcport]}}}" }
    }
  }

  if [dst][port] or [dest][port] or [dstport] {
    mutate {
      add_field => { "dst_port" => "%{[dst][port]:%{[dest][port]:%{[dstport]}}}" }
    }
  }

  # 5. 用户名提取（认证告警）
  if [actor][user][name] or [user][name] or [username] {
    mutate {
      add_field => { "user_name" => "%{[actor][user][name]:%{[user][name]:%{[username]}}}" }
    }
  }

  # 6. 设备主机名
  if [device][hostname] or [hostname] {
    mutate {
      add_field => { "device_hostname" => "%{[device][hostname]:%{[hostname]}}" }
    }
  }

  # 7. 产品信息
  if [product][name] or [product_name] {
    mutate {
      add_field => { "product_name" => "%{[product][name]:%{[product_name]}}" }
    }
  }

  if [product][vendor] or [vendor] {
    mutate {
      add_field => { "product_vendor" => "%{[product][vendor]:%{[vendor]}}" }
    }
  }
```

说明：
- 使用 mutate add_field 进行字段重命名（支持多版本字段别名）
- 使用 ruby 进行告警级别到 severity_id 的数值映射
- 所有新字段添加到事件顶层，便于后续处理
</action>

- filter 部分包含 src_ip、dst_ip 字段提取
- filter 部分包含 alert_title、alert_desc 字段提取
- 使用 ruby filter 进行 severity_id 数值映射
  </acceptance_criteria>

在 filter 部分添加 OCSF 格式结构，为输出到 Kafka 准备标准化数据。

**在字段映射 filter 之后添加 OCSF 结构化 filter:**

```ruby
  # ========== OCSF 格式转换 ==========
  # 参考 OCSF Security Finding (Class 2001) 结构

  mutate {
    add_field => {
      "category_uid" => 2
      "class_uid" => 2001
      "type_uid" => 200101
      "activity_name" => "Security Finding"
    }
  }

  # OCSF time 字段（Unix Epoch 毫秒）
  ruby {
    code => "
      ts = event.get('@timestamp')
      if ts
        begin
          require 'time'
          epoch_ms = Time.parse(ts.to_s).to_i * 1000
          event.set('time', epoch_ms)
        rescue
          event.set('time', nil)
        end
      end
    "
  }

  # severity 字段标准化
  mutate {
    add_field => { "severity" => "%{alert_level}" }
  }

  # status 字段（默认 New）
  mutate {
    add_field => {
      "status_id" => 1
      "status" => "New"
    }
  }

  # finding_info 对象
  ruby {
    code => "
      title = event.get('alert_title') || 'Unknown Alert'
      desc = event.get('alert_desc') || ''
      event.set('finding_info', {
        'title' => title,
        'desc' => desc
      })
    "
  }

  # observables 可观察对象数组（提取关键 IOC）
  ruby {
    code => "
      observables = []
      # IP 地址
      ['src_ip', 'dst_ip'].each do |ip_field|
        ip = event.get(ip_field)
        if ip && ip != 'unknown'
          observables << { 'type_id' => 1, 'type' => 'IP Address', 'value' => ip }
        end
      end
      # 用户名
      user = event.get('user_name')
      if user && user != 'unknown'
        observables << { 'type_id' => 2, 'type' => 'Account', 'value' => user }
      end
      event.set('observables', observables) if observables.any?
    "
  }

  # device 对象
  ruby {
    code => "
      device = {}
      hostname = event.get('device_hostname')
      device['hostname'] = hostname if hostname && hostname != 'unknown'
      event.set('device', device)
    "
  }

  # metadata 产品信息
  ruby {
    code => "
      metadata = { 'product' => {} }
      product_name = event.get('product_name')
      product_vendor = event.get('product_vendor')
      metadata['product']['name'] = product_name if product_name && product_name != 'unknown'
      metadata['product']['vendor_name'] = product_vendor if product_vendor && product_vendor != 'unknown'
      event.set('metadata', metadata)
    "
  }

  # 清理中间临时字段（只保留 OCSF 标准字段）
  mutate {
    remove_field => ["alert_level", "alert_title", "alert_desc", "src_ip", "dst_ip", "src_port", "dst_port", "user_name", "device_hostname", "product_name", "product_vendor"]
  }
```

说明：
- OCSF category_uid=2 表示 Detective Control / Findings
- class_uid=2001 表示 Security Finding
- type_uid=200101 = 2001*100 + 1 (activity_id=1)
- observables 数组提取关键 IOC（IP、用户名）便于威胁情报匹配
- 最后清理中间临时字段，只保留 OCSF 标准结构
</action>

- filter 部分包含 time 字段（Unix Epoch 毫秒）
- filter 部分包含 finding_info 对象
- filter 部分包含 observables 数组
- filter 包含 mutate remove_field 清理临时字段
  </acceptance_criteria>
